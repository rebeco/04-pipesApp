Pipe date

Pipe para formatear fechas

  • fecha = new Date()
  • fecha | date (por defecto formato "mediumDate")
  • fecha | date: "short"
  • fecha | date: "long"
  • fecha | date: "dd/MM/yyy HH:mm"
  • (zona horaria) fecha | date: "long":"GMT-2"
  • (locale) fecha | date: "long":"GMT-2":"en"
  • {{ fecha }}
  • {{ fecha | date }}
  • {{ fecha | date: "short" }}
  • {{ fecha | date: "long" }}
  • {{ fecha | date: "dd/MM/yyy HH:mm" }}
  • {{ fecha | date: "long":"GMT-2" }}
  • {{ fecha | date: "long":"GMT-2":"en" }}
